(1)promise

2021-04-12

PromiseA+

states

  1. pending
    initial state, can change into fulfilled through resolve / rejected through reject
  2. fullfilled
    final state, stable, must contain value
  3. rejected
    final state, stable, must contain reason
    pending -> resolve(value) -> fulfilled
    pending -> reject(reason) -> rejected

then

promise.then(onFulfilled, onRejected)

  1. onFulfilled and onRejected muse be function, or it would be ignored

  2. only after promise’s state change into fulfilled, onFulfilled can callback for 1 time with parameter ‘value’

  3. only after promise’s state change into rejected, onRejected can callback for 1 time with parameter ‘reason’

  4. onFulfilled and onRejected must be microtask (use queueMicrotask)

  5. ‘then’ can be called for N times
    5.1 After promise’s state change into fulfilled, all of onFulfilled should be called following the regist order (require array for saving the sort of onFulfilled)
    5.2 onRejected like the same way

  6. ‘then’ should return a new promise

    promise2 = promise1.then(onFulfilled, onRejected)

    6.1 if onFulfilled get result(marked as x), execute resolvePromise
    6.2 if onFulfilled throw exception, promise2 must be reject

    6.3 if onFulfilled is not a function, promise2 do nothing, only change into fulfilled with the value of promise1
    6.4 onRejected like the same way

  7. resolvePromise

    resolvePromise(proimse2, x, resolve, reject)

    7.1 if promise2 == x, reject TypeError
    7.2 if x is a promise

     if x is pending, promise must be pending until x 'status change
 if x change into fulfilled, fulfill promise with the same value
 if x change into rejected, reject promise with the same reason

    7.3 if x is an object or function

    let then = x.then

     if x.then throw error, reject promise with e as reason
 if 'then' is function, then.call(x, resolvePromiseFn, rejectPromise)
     resolvePromiseFn's parameter is y
         resolvePromise(promise2, y, resolve, reject)

     resolvePromise's paramter is r
         reject promise with r

     if resolvePromise and rejectPromise both called
         the first called execute and the other would be ignored
     if 'then' throwed exception, 
         if resolvePromise or rejectPromise already be called, and should be ignored
 if then is not function, fulfill promise with x

imp

const PENDING = 'pending'
const FULFILLED = 'fulfilled'
const REJECTED = 'rejected'

class MPromise {
  FULFILLED_CALLBACK_LIST = []
  REJECTED_CALLBACK_LIST = []

  _status = PENDING

  constructor(fn) {
    this.status = PENDING
    this.value = null
    this.reason = null

    // fn execute immediately, with 2 parameters, if get error should  throw exception
    try {
      // 'bind' make sure 'this' point at the right way.
      fn(this.resolve.bind(this), this.reject.bind(this))
    } catch (e) {
      this.reject(e)
    }
  }

  get status() {
    return this._status
  }

  set status(newStatus) {
    this._status = newStatus
    switch (newStatus) {
      case FULFILLED: {
        this.FULFILLED_CALLBACK_LIST.forEach((callback) => callback(this.value))
        break
      }
      case REJECTED: {
        this.REJECTED_CALLBACK_LIST.forEach((callback) => callback(this.reason))
        break
      }
    }
  }

  resolve(value) {
    if (this.status === PENDING) {
      this.value = value
      this.status = FULFILLED
    }
  }

  reject(reason) {
    if (this.status === PENDING) {
      this.reason = reason
      this.status = REJECTED
    }
  }

  then(onFulfilled, onRejected) {
    const fulFilledFn = this.isFunction(onFulfilled)
      ? onFulfilled
      : (value) => value
    const rejectedFn = this.isFunction(onRejected)
      ? onRejected
      : (reason) => reason

    // define function with catch
    const fulFilledFnWithCatch = (resolve, reject) => {
      queueMicrotask(() => {
        try {
          if (!this.isFunction(onFulfilled)) {
            resolve(this.value)
          } else {
            const x = fulFilledFn(this.value)
            this.resolvePromise(newPromise, x, resolve, reject)
          }
          resolve(this.value)
        } catch (e) {
          reject(e)
        }
      })
    }

    const rejectedFnWithCatch = (resolve, reject, newPromise) => {
      queueMicrotask(() => {
        try {
          if (!this.isFunction(onRejected)) {
            reject(this.reason)
          } else {
            const x = rejectedFn(this.reason)
            this.resolvePromise(newPromise, x, resolve, reject)
          }
        } catch (e) {
          reject(e)
        }
      })
    }

    switch (this.status) {
      case FULFILLED: {
        const newPromise = new MPromise((resolve, reject) =>
          fulFilledFnWithCatch(resolve, reject, newPromise)
        )
        return newPromise
      }
      case REJECTED: {
        const newPromise = new MPromise((resolve, reject) =>
          rejectedFnWithCatch(resolve, reject, newPromise)
        )
        return newPromise
      }
      case PENDING: {
        const newPromise = new MPromise((resolve, reject) => {
          this.FULFILLED_CALLBACK_LIST.push(() =>
            fulFilledFnWithCatch(resolve, reject)
          )
          this.REJECTED_CALLBACK_LIST.push(() =>
            rejectedFnWithCatch(resolve, reject)
          )
        })
        return newPromise
      }
    }
  }

  catch (onRejected) {
    return this.then(null, onRejected)
  }

  resolvePromise(newPromise, x, resolve, reject) {
    if (newPromise === x) {
      return reject(new TypeError('error'))
    }
    if (x instanceof MPromise) {
      x.then((y) => {
        this.resolvePromise(newPromise(newPromise, y, resolve, reject))
      }, reject)
    } else if (typeof x === 'object' || this.isFunction(x)) {
      if (x === null) {
        return resolve(x)
      }

      let then = null
      try {
        then = x.then
      } catch (error) {
        return reject(error)
      }

      if (this.isFunction(then)) {
        let called = false
        try {
          then.call(
            x,
            (y) => {
              if (called) {
                return
              }
              called = true
              this.resolvePromise(newPromise, y, resolve, reject)
            },
            (r) => {
              if (called) {
                return
              }
              called = true
              reject(r)
            }
          )
        } catch (error) {
          if (called) {
            return
          }
          reject(error)
        }
      } else {
      }
    } else {
      resolve(x)
    }
  }

  isFunction(param) {
    return typeof param === 'function'
  }
}

const test = new MPromise((resolve, reject) => {
  setTimeout(() => {
    // resolve(1111)
    reject(2222)
  }, 1000)
}).then((value) => {
  console.log('完成' + value)
  console.log(test)
}).catch((reason) => {
  console.log('error' + reason)
  console.log(test)
})
