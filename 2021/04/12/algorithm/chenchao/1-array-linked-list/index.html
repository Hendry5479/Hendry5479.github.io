## 283. move-zeroes 2
移动零元素到数组末尾
+ 输入: [0,1,0,3,12]
+ 输出: [1,3,12,0,0]
``` javascript
var moveZeroes = function(nums) {
  // 双指针。一个指针记录当前遍历的位置，一个用来记录最前面的非零位置
  let len = nums.length
  let index = 0

  for (let i = 0; i < len; i++) {
    if (nums[i] == 0)
      continue
    if (index == i) {
      index ++
      continue
    }
    nums[index] = nums[i]
    index ++
    nums[i] = 0
  }
  return nums
};

70. climbing-stairs

n 阶,每次爬 1 或 2 个台阶,有多少种不同的方法爬到楼顶

公式 dp[n] = dp[n-1] + dp[n-2]dp[n]=dp[n−1]+dp[n−2]

var climbStairs = function(n) {
    const dp = [];
    dp[0] = 1;
    dp[1] = 1;
    for(let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
};

15. 3sum

包含 n 个整数的数组 nums，找出所有和为 0 且不重复的三元组

注意：答案中不可以包含重复的三元组。

var threeSum = function(nums) {
  let size = nums.length
  // 小于3返回空
  if (size < 3) {
    return []
  }
  res = []

  // 排序
  nums.sort((a, b) => { return a - b})
  
  for (let i = 0; i < size; i++) {
    // 三数中的最小值一定要小于零
    if (nums[i] > 0) {
      return res
    }
    // nums[i]与[i-1]相等，说明即使找到也是重复的三元组，跳过
    if (i > 0 && nums[i] == nums[i-1]) {
      continue
    }
    // 从i后面的元素找到两个目标元素
    let left = i+1, right = size - 1
    while (left < right) {
      if (nums[left] + nums[right] + nums[i] == 0) {
        res.push([nums[i], nums[left], nums[right]])
        // 跳过重复元素
        while (left < right && nums[left] == nums[left + 1]) {
          left += 1
        }
        while (left < right && nums[right] == nums[right - 1]) {
          right += 1
        }
        // 找到后两指针各位移一次
        left += 1
        right -= 1
      } else if (nums[left] + nums[right] + nums[i] > 0) {
        right -= 1
      } else {
        left += 1
      }
    }
  }
  return res
};

206. Reverse Linked List

// 双指针
var reverseList = function(head) {
  let cur = null, pre = head
  while (pre != null) {
    
    let temp = pre.next
    pre.next = cur
    cur = pre
    pre = temp
  }
  return cur
};

24. Swap Nodes in Pairs

两两交换其中相邻的节点，并返回交换后的链表，需要实际进行节点交换

输入：head = [1,2,3,4]
输出：[2,1,4,3]

var swapPairs = function(head) {
  // 新建空节点，指向头节点
  let pre = new ListNode(0)
  pre.next = head 

  // 设置temp为反转节点的前一个节点
  let temp = pre
  while (temp.next != null && temp.next.next != null) {
    let start = temp.next
    let end = temp.next.next
    // 将start和end互换位置
    temp.next = end
    start.next = end.next
    end.next = start
    // temp位移
    temp = start
  }
  return pre.next
}
