11. container-with-most-water

class Solution:
    # 76ms, 击败了53.85 % 双指针
    def maxArea(self, height: List[int]) -> int:
        max_area = 0
        left, right = 0, len(height) - 1
        while left < right:
            width = right - left
            if height[left] >= height[right]:
                area = height[right] * width
                right -= 1
            else:
                area = height[left] * width
                left += 1
            max_area = max(max_area, area)
        return max_area
    # 参考1， 100 ms,击败了8.83%  双指针  Krahets
    def maxArea(self, height: List[int]) -> int:
        i, j, res = 0, len(height) - 1, 0
        while i < j:
            if height[i] < height[j]:
                res = max(res, height[i] * (j - i))
                i += 1
            else:
                res = max(res, height[j] * (j - i))
                j -= 1
        return res

var isValid = function(s) {
  let stack = [], len = s.length

  for (let i = 0; i < len; i++) {
    if (stack.length == 0 && (s[i] == '}' || s[i] == ')' || s[i] == ']')) {
      return false
    }
    if (s[i] == '{' || s[i] == '(' || s[i] == '[') {
      stack.push(s[i])
    } else if ((s[i] == ')' && stack[stack.length-1] == '(') || 
              (s[i] == '}' && stack[stack.length-1] == '{') ||
              (s[i] == ']' && stack[stack.length-1] == '[')
    ) {
      stack.pop()
    } else {
      return false
    }
  }
  // 遍历完还有未匹配括号
  if (stack.length !== 0) {
    return false
  }
  return true
};

    # 栈 + 字典  36 ms,击败了89.71%
    def isValid(self, s: str) -> bool:
        stack = []
        mapping = {')': '(', ']': '[', '}': '{'}
        for char in s:
            if char in mapping:
                top_element = stack.pop() if stack else '#'
                if mapping[char] != top_element:
                    return False
            else:
                stack.append(char)
        return not stack

155. Min Stack

    # 借助辅助站min_stack，保存stack最小值
    # push()，如果 小于等于 min_stack栈顶，则一起push()即更新栈顶最小值
    # pop()，判断元素是否是min_stack栈顶，如果是则将min_stack栈顶元素一起pop()
    # getMin()： 返回min_stack栈顶即可
    def __init__(self):
        self.stack = []
        self.min_stack = [math.inf]

    def push(self, x: int) -> None:
        self.stack.append(x)
        self.min_stack.append(min(x, self.min_stack[-1]))

    def pop(self) -> None:
        self.stack.pop()
        self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]

84. Largest Rectangle in Histogram

代码 1 需要考虑两种特殊的情况：
1.弹栈的时候，栈为空；
2.遍历完成以后，栈中还有元素；
为此在输入数组的两端加上两个高度为 0 （或者是 0.5，只要比 1 严格小）的柱形，回避讨论。
这两个站在两边的柱形叫做哨兵（Sentinel）：
左边的柱形：比任何元素小，肯定不会出栈，因此栈一定不为空；
右边的柱形：比任何元素小，会让所有元素出栈（哨兵除外）

    # 单调递增栈
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        stack = []

        for i in range(size):
            while len(stack) > 0 and heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                while len(stack) > 0 and cur_height == heights[stack[-1]]:
                    stack.pop()

                if len(stack) > 0:
                    cur_width = i - stack[-1] - 1
                else:
                    cur_width = i
                res = max(cur_width * cur_height, res)
            stack.append(i)

        while len(stack) > 0:
            cur_height = heights[stack.pop()]
            while len(stack) > 0 and cur_height == heights[stack[-1]]:
                stack.pop()

            if len(stack) > 0:
                cur_width = size - stack[-1] - 1
            else:
                cur_width = size
            res = max(cur_width * cur_height, res)

        return res

    # 单调栈 + 哨兵
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        heights = [0] + heights + [0]
        # 先放入哨兵结点，在循环中就不用做非空判断
        stack = [0]
        size += 2
        for i in range(1, size):
            while heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                cur_width = i - stack[-1] - 1
                res = max(res, cur_height * cur_width)
            stack.append(i)
        return res

239. Sliding Window Maximum

单调递减队列，队头是最大值
pop：如果窗口弹出的是队头，则将队头删去
push：如果窗口进入的值 > 队尾，则将队尾删去，直至队尾大于窗口的新元素
front：返回最大值，即队头
https://leetcode-cn.com/problems/sliding-window-maximum/solution/3chong-jie-jue-fang-shi-by-sdwwld/

    # 单调队列 300ms 47%
    def maxSlidingWindow1(self, nums: List[int], k: int) -> List[int]:
        que = deque()
        n = len(nums)
        for i in range(k):
            while que and que[-1] < nums[i]:
                que.pop()
            que.append(nums[i])
        res = [que[0]]
        for i in range(k, n):
            if que[0] == nums[i - k]:
                del que[0]
            while que and que[-1] < nums[i]:
                que.pop()
            que.append(nums[i])
            res.append(que[0])
        return res

641. Design Circular Deque

class MyCircularDeque:
    def __init__(self, k: int):
        """
        Initialize your data structure here. Set the size of the deque to be k.
        """
        self.max_len = k
        self.queue = []

    def insertFront(self, value: int) -> bool:
        """
        Adds an item at the front of Deque. Return true if the operation is successful.
        """
        if len(self.queue) < self.max_len:
            self.queue = self.queue[::-1]  # [::-1]使列表倒序
            self.queue.append(value)
            self.queue = self.queue[::-1]
            return True
        return False

    def insertLast(self, value: int) -> bool:
        """
        Adds an item at the rear of Deque. Return true if the operation is successful.
        """
        if len(self.queue) < self.max_len:
            self.queue.append(value)
            return True
        return False

    def deleteFront(self) -> bool:
        """
        Deletes an item from the front of Deque. Return true if the operation is successful.
        """
        if len(self.queue):
            del self.queue[0]
            return True
        return False

    def deleteLast(self) -> bool:
        """
        Deletes an item from the rear of Deque. Return true if the operation is successful.
        """
        if len(self.queue):
            del self.queue[-1]
            return True
        return False

    def getFront(self) -> int:
        """
        Get the front item from the deque.
        """
        if len(self.queue):
            return self.queue[0]
        return -1

    def getRear(self) -> int:
        """
        Get the last item from the deque.
        """
        if len(self.queue):
            return self.queue[-1]
        return -1

    def isEmpty(self) -> bool:
        """
        Checks whether the circular deque is empty or not.
        """
        return not len(self.queue)

    def isFull(self) -> bool:
        """
        Checks whether the circular deque is full or not.
        """
        return len(self.queue) == self.max_len

42. Trapping Rain Water

    # 双指针法 40 ms,击败了94.74%
    def trap1(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        res = 0
        left_highest = height[0]
        right_highest = height[size-1]
        left = 1
        right = size - 2
        while (left <= right):
            if left_highest < right_highest:
                if height[left] < left_highest:
                    res += (left_highest - height[left])
                else:
                    left_highest = height[left]
                left += 1
            else:
                if height[right] < right_highest:
                    res += (right_highest - height[right])
                else:
                    right_highest = height[right]
                right -= 1
        return res

    # DP 52 ms,击败了54.18%
    def trap2(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        res = 0
        left_highest = [0 for i in range(size)]
        left_highest[0] = height[0]

        right_highest = [0 for i in range(size)]
        right_highest[size-1] = height[size-1]

        for i in range(1, size-1):
            left_highest[i] = max(height[i-1], left_highest[i-1])
        for i in range(size-2, 0, -1):
            right_highest[i] = max(height[i+1], right_highest[i+1])
        for i in range(1, size-1):
            min_height = min(left_highest[i], right_highest[i])
            if min_height > height[i]:
                res += min_height - height[i]
        return res

    # stack 60 ms,击败了23.45%
    def trap3(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        stack = [0]
        res = 0
        for i in range(size):
            if height[stack[-1]] > height[i]:
                stack.append(i)
            elif height[stack[-1]] == height[i]:
                stack[-1] = i
            else:
                while len(stack) != 0 and height[stack[-1]] < height[i]:
                    mid = stack.pop()
                    if len(stack) != 0:
                        min_height = min(height[stack[-1]], height[i]) - height[mid]
                        res += min_height * (i - stack[-1] - 1)
                stack.append(i)
        return res
