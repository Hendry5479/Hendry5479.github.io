  // 双指针。一个指针记录当前遍历的位置，一个用来记录最前面的非零位置
  let len = nums.length

var climbStairs = function(n) {
    const dp = [];
    dp[0] = 1;
    dp[1] = 1;
    for(let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
};

    def maxArea(self, height: List[int]) -> int:
        max_area = 0
        left, right = 0, len(height) - 1
        while left < right:
            width = right - left
            if height[left] >= height[right]:
                area = height[right] * width
                right -= 1
            else:
                area = height[left] * width
                left += 1
            max_area = max(max_area, area)
        return max_area

        i, j, res = 0, len(height) - 1, 0
        while i < j:
            if height[i] < height[j]:
                res = max(res, height[i] * (j - i))
                i += 1
            else:
                res = max(res, height[j] * (j - i))
                j -= 1
        return res

var isValid = function(s) {
  let stack = [], len = s.length

  for (let i = 0; i < len; i++) {
    if (stack.length == 0 && (s[i] == '}' || s[i] == ')' || s[i] == ']')) {
      return false
    }
    if (s[i] == '{' || s[i] == '(' || s[i] == '[') {
      stack.push(s[i])
    } else if ((s[i] == ')' && stack[stack.length-1] == '(') || 
              (s[i] == '}' && stack[stack.length-1] == '{') ||
              (s[i] == ']' && stack[stack.length-1] == '[')
    ) {
      stack.pop()
    } else {
      return false
    }
  }
  // 遍历完还有未匹配括号
  if (stack.length !== 0) {
    return false
  }
  return true
};

    def isValid(self, s: str) -> bool:
        stack = []
        mapping = {')': '(', ']': '[', '}': '{'}
        for char in s:
            if char in mapping:
                top_element = stack.pop() if stack else '#'
                if mapping[char] != top_element:
                    return False
            else:
                stack.append(char)
        return not stack

155. Min Stack

    # 借助辅助站min_stack，保存stack最小值
    # push()，如果 小于等于 min_stack栈顶，则一起push()即更新栈顶最小值
    # pop()，判断元素是否是min_stack栈顶，如果是则将min_stack栈顶元素一起pop()
    # getMin()： 返回min_stack栈顶即可
    def __init__(self):
        self.stack = []
        self.min_stack = [math.inf]

    def push(self, x: int) -> None:
        self.stack.append(x)
        self.min_stack.append(min(x, self.min_stack[-1]))

    def pop(self) -> None:
        self.stack.pop()
        self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]

84. Largest Rectangle in Histogram

代码 1 需要考虑两种特殊的情况：
1.弹栈的时候，栈为空；
2.遍历完成以后，栈中还有元素；
为此在输入数组的两端加上两个高度为 0 （或者是 0.5，只要比 1 严格小）的柱形，回避讨论。
这两个站在两边的柱形叫做哨兵（Sentinel）：
左边的柱形：比任何元素小，肯定不会出栈，因此栈一定不为空；
右边的柱形：比任何元素小，会让所有元素出栈（哨兵除外）

    # 单调递增栈
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        stack = []

        for i in range(size):
            while len(stack) > 0 and heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                while len(stack) > 0 and cur_height == heights[stack[-1]]:
                    stack.pop()

                if len(stack) > 0:
                    cur_width = i - stack[-1] - 1
                else:
                    cur_width = i
                res = max(cur_width * cur_height, res)
            stack.append(i)

        while len(stack) > 0:
            cur_height = heights[stack.pop()]
            while len(stack) > 0 and cur_height == heights[stack[-1]]:
                stack.pop()

            if len(stack) > 0:
                cur_width = size - stack[-1] - 1
            else:
                cur_width = size
            res = max(cur_width * cur_height, res)

        return res

    # 单调栈 + 哨兵
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        heights = [0] + heights + [0]
        # 先放入哨兵结点，在循环中就不用做非空判断
        stack = [0]
        size += 2
        for i in range(1, size):
            while heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                cur_width = i - stack[-1] - 1
                res = max(res, cur_height * cur_width)
            stack.append(i)
        return res

30.设计模式

设计模式总的来说是一个抽象的概念，前人通过无数次的实践总结出的一套写代码的方式，通过这种方式写的代码可以让别人更加容易阅读、维护以及复用。

简单工厂模式

class Man {
  constructor(name) {
    this.name = name
  }
  alertName() {
    alert(this.name)
  }
}

class Factory {
  static create(name) {
    return new Man(name)
  }
}

Factory.create('yck').alertName()

当然工厂模式并不仅仅是用来 new 出实例。

可以想象一个场景。假设有一份很复杂的代码需要用户去调用，但是用户并不关心这些复杂的代码，只需要你提供给我一个接口去调用，用户只负责传递需要的参数，至于这些参数怎么使用，内部有什么逻辑是不关心的，只需要你最后返回我一个实例。这个构造过程就是工厂。

工厂起到的作用就是隐藏了创建实例的复杂度，只需要提供一个接口，简单清晰。

在 Vue 源码中，你也可以看到工厂模式的使用，比如创建异步组件

29.输入 URL 到页面渲染的整个流程

大致过程

DNS解析、TCP握手建立连接、TLS握手开始传输数据、浏览器判断状态码、解析文件；
构建DOM树和CSSOM树，遇到script标签就去请求js文件并执行，生成Render树，调用GPU绘制

详细过程

首先是 DNS 查询，如果这一步做了智能 DNS 解析的话，会提供访问速度最快的 IP 地址回来，这部分的内容之前没有写过，所以就在这里讲解下。

DNS

DNS 的作用就是通过域名查询到具体的 IP。

因为 IP 存在数字和英文的组合（IPv6），很不利于人类记忆，所以就出现了域名。你可以把域名看成是某个 IP 的别名，DNS 就是去查询这个别名的真正名称是什么。

在 TCP 握手之前就已经进行了 DNS 查询，这个查询是操作系统自己做的。当你在浏览器中想访问 www.google.com 时，会进行一下操作：

操作系统会首先在本地缓存中查询 IP
没有的话会去系统配置的 DNS 服务器中查询
如果这时候还没得话，会直接去 DNS 根服务器查询，这一步查询会找出负责 com 这个一级域名的服务器
然后去该服务器查询 google 这个二级域名
接下来三级域名的查询其实是我们配置的，你可以给 www 这个域名配置一个 IP，然后还可以给别的三级域名配置一个 IP
以上介绍的是 DNS 迭代查询，还有种是递归查询，区别就是前者是由客户端去做请求，后者是由系统配置的 DNS 服务器做请求，得到结果后将数据返回给客户端。
