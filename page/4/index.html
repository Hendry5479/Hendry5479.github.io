var moveZeroes = function(nums) {
  // 双指针。一个指针记录当前遍历的位置，一个用来记录最前面的非零位置
  let len = nums.length
  let index = 0

  for (let i = 0; i < len; i++) {
    if (nums[i] == 0)

    if (index == i) {
      index ++

var climbStairs = function(n) {
    const dp = [];
    dp[0] = 1;
    dp[1] = 1;
    for(let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];

15. 3sum

包含 n 个整数的数组 nums，找出所有和为 0 且不重复的三元组

注意：答案中不可以包含重复的三元组。

var threeSum = function(nums) {
  let size = nums.length
  // 小于3返回空
  if (size < 3) {
    return []
  }
  res = []

  // 排序
  nums.sort((a, b) => { return a - b})
  
  for (let i = 0; i < size; i++) {
    // 三数中的最小值一定要小于零
    if (nums[i] > 0) {
      return res
    }
    // nums[i]与[i-1]相等，说明即使找到也是重复的三元组，跳过
    if (i > 0 && nums[i] == nums[i-1]) {
      continue
    }
    // 从i后面的元素找到两个目标元素
    let left = i+1, right = size - 1
    while (left < right) {
      if (nums[left] + nums[right] + nums[i] == 0) {
        res.push([nums[i], nums[left], nums[right]])
        // 跳过重复元素
        while (left < right && nums[left] == nums[left + 1]) {
          left += 1
        }
        while (left < right && nums[right] == nums[right - 1]) {
          right += 1
        }
        // 找到后两指针各位移一次
        left += 1
        right -= 1
      } else if (nums[left] + nums[right] + nums[i] > 0) {
        right -= 1
      } else {
        left += 1
      }
    }
  }
  return res
};

var swapPairs = function(head) {
  // 新建空节点，指向头节点
  let pre = new ListNode(0)
  pre.next = head 

  // 设置temp为反转节点的前一个节点
  let temp = pre
  while (temp.next != null && temp.next.next != null) {
    let start = temp.next
    let end = temp.next.next
    // 将start和end互换位置
    temp.next = end
    start.next = end.next
    end.next = start
    // temp位移
    temp = start
  }
  return pre.next
}

155. Min Stack

    # 借助辅助站min_stack，保存stack最小值
    # push()，如果 小于等于 min_stack栈顶，则一起push()即更新栈顶最小值
    # pop()，判断元素是否是min_stack栈顶，如果是则将min_stack栈顶元素一起pop()
    # getMin()： 返回min_stack栈顶即可
    def __init__(self):
        self.stack = []
        self.min_stack = [math.inf]

    def push(self, x: int) -> None:
        self.stack.append(x)
        self.min_stack.append(min(x, self.min_stack[-1]))

    def pop(self) -> None:
        self.stack.pop()
        self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]

代码 1 需要考虑两种特殊的情况：
1.弹栈的时候，栈为空；
2.遍历完成以后，栈中还有元素；
为此在输入数组的两端加上两个高度为 0 （或者是 0.5，只要比 1 严格小）的柱形，回避讨论。
这两个站在两边的柱形叫做哨兵（Sentinel）：
左边的柱形：比任何元素小，肯定不会出栈，因此栈一定不为空；
右边的柱形：比任何元素小，会让所有元素出栈（哨兵除外）

    # 单调递增栈
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        stack = []

        for i in range(size):
            while len(stack) > 0 and heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                while len(stack) > 0 and cur_height == heights[stack[-1]]:
                    stack.pop()

                if len(stack) > 0:
                    cur_width = i - stack[-1] - 1
                else:
                    cur_width = i
                res = max(cur_width * cur_height, res)
            stack.append(i)

        while len(stack) > 0:
            cur_height = heights[stack.pop()]
            while len(stack) > 0 and cur_height == heights[stack[-1]]:
                stack.pop()

            if len(stack) > 0:
                cur_width = size - stack[-1] - 1
            else:
                cur_width = size
            res = max(cur_width * cur_height, res)

        return res

    # 单调栈 + 哨兵
    def largestRectangleArea(self, heights: List[int]) -> int:
        size = len(heights)
        res = 0
        heights = [0] + heights + [0]
        # 先放入哨兵结点，在循环中就不用做非空判断
        stack = [0]
        size += 2
        for i in range(1, size):
            while heights[i] < heights[stack[-1]]:
                cur_height = heights[stack.pop()]
                cur_width = i - stack[-1] - 1
                res = max(res, cur_height * cur_width)
            stack.append(i)
        return res

    def maxSlidingWindow1(self, nums: List[int], k: int) -> List[int]:
        que = deque()
        n = len(nums)
        for i in range(k):
            while que and que[-1] < nums[i]:
                que.pop()
            que.append(nums[i])
        res = [que[0]]

                del que[0]
            while que and que[-1] < nums[i]:
                que.pop()
            que.append(nums[i])
            res.append(que[0])

641. Design Circular Deque

class MyCircularDeque:
    def __init__(self, k: int):
        """
        Initialize your data structure here. Set the size of the deque to be k.
        """
        self.max_len = k
        self.queue = []

    def insertFront(self, value: int) -> bool:
        """
        Adds an item at the front of Deque. Return true if the operation is successful.
        """
        if len(self.queue) < self.max_len:
            self.queue = self.queue[::-1]  # [::-1]使列表倒序
            self.queue.append(value)
            self.queue = self.queue[::-1]
            return True
        return False

    def insertLast(self, value: int) -> bool:
        """
        Adds an item at the rear of Deque. Return true if the operation is successful.
        """
        if len(self.queue) < self.max_len:
            self.queue.append(value)
            return True
        return False

    def deleteFront(self) -> bool:
        """
        Deletes an item from the front of Deque. Return true if the operation is successful.
        """
        if len(self.queue):
            del self.queue[0]
            return True
        return False

    def deleteLast(self) -> bool:
        """
        Deletes an item from the rear of Deque. Return true if the operation is successful.
        """
        if len(self.queue):
            del self.queue[-1]
            return True
        return False

    def getFront(self) -> int:
        """
        Get the front item from the deque.
        """
        if len(self.queue):
            return self.queue[0]
        return -1

    def getRear(self) -> int:
        """
        Get the last item from the deque.
        """
        if len(self.queue):
            return self.queue[-1]
        return -1

    def isEmpty(self) -> bool:
        """
        Checks whether the circular deque is empty or not.
        """
        return not len(self.queue)

    def isFull(self) -> bool:
        """
        Checks whether the circular deque is full or not.
        """
        return len(self.queue) == self.max_len

    def trap1(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        res = 0
        left_highest = height[0]
        right_highest = height[size-1]
        left = 1
        right = size - 2

            if left_highest < right_highest:
                if height[left] < left_highest:

                else:
                    left_highest = height[left]
                left += 1
            else:

                else:
                    right_highest = height[right]
                right -= 1
        return res

    def trap2(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        res = 0
        left_highest = [0 for i in range(size)]
        left_highest[0] = height[0]

        right_highest = [0 for i in range(size)]
        right_highest[size-1] = height[size-1]

        for i in range(1, size-1):
            left_highest[i] = max(height[i-1], left_highest[i-1])
        for i in range(size-2, 0, -1):
            right_highest[i] = max(height[i+1], right_highest[i+1])
        for i in range(1, size-1):
            min_height = min(left_highest[i], right_highest[i])
            if min_height > height[i]:
                res += min_height - height[i]
        return res

    def trap3(self, height: List[int]) -> int:
        size = len(height)
        if size < 3:
            return 0
        stack = [0]
        res = 0
        for i in range(size):
            if height[stack[-1]] > height[i]:
                stack.append(i)
            elif height[stack[-1]] == height[i]:
                stack[-1] = i
            else:

                    mid = stack.pop()
                    if len(stack) != 0:

                stack.append(i)
        return res

PromiseA+

states

  1. pending
    initial state, can change into fulfilled through resolve / rejected through reject
  2. fullfilled
    final state, stable, must contain value
  3. rejected
    final state, stable, must contain reason
    pending -> resolve(value) -> fulfilled
    pending -> reject(reason) -> rejected

then

promise.then(onFulfilled, onRejected)

  1. onFulfilled and onRejected muse be function, or it would be ignored

  2. only after promise’s state change into fulfilled, onFulfilled can callback for 1 time with parameter ‘value’

  3. only after promise’s state change into rejected, onRejected can callback for 1 time with parameter ‘reason’

  4. onFulfilled and onRejected must be microtask (use queueMicrotask)

  5. ‘then’ can be called for N times
    5.1 After promise’s state change into fulfilled, all of onFulfilled should be called following the regist order (require array for saving the sort of onFulfilled)
    5.2 onRejected like the same way

  6. ‘then’ should return a new promise

    promise2 = promise1.then(onFulfilled, onRejected)

    6.1 if onFulfilled get result(marked as x), execute resolvePromise
    6.2 if onFulfilled throw exception, promise2 must be reject

    6.3 if onFulfilled is not a function, promise2 do nothing, only change into fulfilled with the value of promise1
    6.4 onRejected like the same way

  7. resolvePromise

    resolvePromise(proimse2, x, resolve, reject)

    7.1 if promise2 == x, reject TypeError
    7.2 if x is a promise

     if x is pending, promise must be pending until x 'status change
 if x change into fulfilled, fulfill promise with the same value
 if x change into rejected, reject promise with the same reason

    7.3 if x is an object or function

    let then = x.then

     if x.then throw error, reject promise with e as reason
 if 'then' is function, then.call(x, resolvePromiseFn, rejectPromise)
     resolvePromiseFn's parameter is y
         resolvePromise(promise2, y, resolve, reject)

     resolvePromise's paramter is r
         reject promise with r

     if resolvePromise and rejectPromise both called
         the first called execute and the other would be ignored
     if 'then' throwed exception, 
         if resolvePromise or rejectPromise already be called, and should be ignored
 if then is not function, fulfill promise with x

const PENDING = 'pending'
const FULFILLED = 'fulfilled'
const REJECTED = 'rejected'

class MPromise {
  FULFILLED_CALLBACK_LIST = []
  REJECTED_CALLBACK_LIST = []

  _status = PENDING

  constructor(fn) {
    this.status = PENDING
    this.value = null
    this.reason = null

    // fn execute immediately, with 2 parameters, if get error should  throw exception
    try {
      // 'bind' make sure 'this' point at the right way.
      fn(this.resolve.bind(this), this.reject.bind(this))
    } catch (e) {
      this.reject(e)
    }
  }

  get status() {
    return this._status
  }

  set status(newStatus) {
    this._status = newStatus
    switch (newStatus) {
      case FULFILLED: {
        this.FULFILLED_CALLBACK_LIST.forEach((callback) => callback(this.value))
        break
      }
      case REJECTED: {
        this.REJECTED_CALLBACK_LIST.forEach((callback) => callback(this.reason))
        break
      }
    }
  }

  resolve(value) {
    if (this.status === PENDING) {
      this.value = value
      this.status = FULFILLED
    }
  }

  reject(reason) {
    if (this.status === PENDING) {
      this.reason = reason
      this.status = REJECTED
    }
  }

  then(onFulfilled, onRejected) {
    const fulFilledFn = this.isFunction(onFulfilled)
      ? onFulfilled
      : (value) => value
    const rejectedFn = this.isFunction(onRejected)
      ? onRejected
      : (reason) => reason

    // define function with catch
    const fulFilledFnWithCatch = (resolve, reject) => {
      queueMicrotask(() => {
        try {
          if (!this.isFunction(onFulfilled)) {
            resolve(this.value)
          } else {
            const x = fulFilledFn(this.value)
            this.resolvePromise(newPromise, x, resolve, reject)
          }
          resolve(this.value)
        } catch (e) {
          reject(e)
        }
      })
    }

    const rejectedFnWithCatch = (resolve, reject, newPromise) => {
      queueMicrotask(() => {
        try {
          if (!this.isFunction(onRejected)) {
            reject(this.reason)
          } else {
            const x = rejectedFn(this.reason)
            this.resolvePromise(newPromise, x, resolve, reject)
          }
        } catch (e) {
          reject(e)
        }
      })
    }

    switch (this.status) {
      case FULFILLED: {
        const newPromise = new MPromise((resolve, reject) =>
          fulFilledFnWithCatch(resolve, reject, newPromise)
        )
        return newPromise
      }
      case REJECTED: {
        const newPromise = new MPromise((resolve, reject) =>
          rejectedFnWithCatch(resolve, reject, newPromise)
        )
        return newPromise
      }
      case PENDING: {
        const newPromise = new MPromise((resolve, reject) => {
          this.FULFILLED_CALLBACK_LIST.push(() =>
            fulFilledFnWithCatch(resolve, reject)
          )
          this.REJECTED_CALLBACK_LIST.push(() =>
            rejectedFnWithCatch(resolve, reject)
          )
        })
        return newPromise
      }
    }
  }

  catch (onRejected) {
    return this.then(null, onRejected)
  }

  resolvePromise(newPromise, x, resolve, reject) {
    if (newPromise === x) {
      return reject(new TypeError('error'))
    }
    if (x instanceof MPromise) {
      x.then((y) => {
        this.resolvePromise(newPromise(newPromise, y, resolve, reject))
      }, reject)
    } else if (typeof x === 'object' || this.isFunction(x)) {
      if (x === null) {
        return resolve(x)
      }

      let then = null
      try {
        then = x.then
      } catch (error) {
        return reject(error)
      }

      if (this.isFunction(then)) {
        let called = false
        try {
          then.call(
            x,
            (y) => {
              if (called) {
                return
              }
              called = true
              this.resolvePromise(newPromise, y, resolve, reject)
            },
            (r) => {
              if (called) {
                return
              }
              called = true
              reject(r)
            }
          )
        } catch (error) {
          if (called) {
            return
          }
          reject(error)
        }
      } else {
      }
    } else {
      resolve(x)
    }
  }

  isFunction(param) {
    return typeof param === 'function'
  }
}

    searchHeroes(term: string): Observable<Hero[]> {
  if (!term.trim()) {
    // if not search term, return empty hero array.
    return of([]);
  }

  return this.http.get<Hero[]>(`${this.heroesUrl}/?name=${term}`).pipe(
    tap(x => x.length ?
       this.log(`found heroes matching "${term}"`) :
       this.log(`no heroes matching "${term}"`)),
    catchError(this.handleError<Hero[]>('searchHeroes', []))
  );
}

    RxJS Subject 类型的 searchTerms

search(term: string): void {
  this.searchTerms.next(term);
}

    Subject 既是Observable对象的数据源，本身也是 Observable
    通过调用next(value) 往 Observable 中推送一些值

    串联 RxJS 操作符

    ngOnInit( ) 往 searchTerms 的处理管道中加入 RxJS 操作符，缩减对 searchHeroes( ) 的调用次数，并返回一个搜索结果的可观察对象（Hero[ ] ）

    this.heroes$ = this.searchTerms.pipe(
  // 传出最终字符串之前，debounceTime(300) 将会等待直至300内无新事件
  debounceTime(300),
  // 只在过滤条件变化时发送请求
  distinctUntilChanged(),
  // 取消丢弃旧的数据，保留最新的
  switchMap((term: string) => this.heroService.searchHeroes(term)),

常见场景和设计方法
内嵌：数组内嵌，减少了关联查询，适合单类描述的属性和不经常变化的属性，如果是拓展属性或者需要单独访问的则不合适
父子引用：
  父引用：存在一对多的情况，放在同一文档中，以数组形式存放
  子引用：存在一对多的情况，反向引用，例如文章里的评论
  数据内容非常多，或者需要单独访问，则不适合内嵌
反范式：
  范式指特定的用法，反范式指不走寻常路
  是否有提升性能的空间，数据提升是否庞大
  先考虑读写比，再考虑反范式
